KiviKova deploys an AI bot that silently joins your video calls, transcribes everything in real-time, generates semantic embeddings, and becomes a voice-enabled RAG agent that anyone in the call can query — live or later.

API at a Glance

// POST /api/agent/join
// Sends the bot into a live meeting
export async function POST(req: Request) {
  const { meetingId, joinUrl } = await req.json();
  const bot = await meetingBotClient.createBot({
    meetingUrl: joinUrl,
    botName: "KiviKova",
    transcriptionOptions: { provider: "meeting_captions" },
  });
  await db.update(meetings)
    .set({ botId: bot.id, status: "active" })
    .where(eq(meetings.id, meetingId));
  return NextResponse.json({ botId: bot.id });
}
// POST /api/webhooks/recall/transcript
// Embeds incoming transcript chunks
export async function POST(req: Request) {
  const { data } = await req.json();
  const chunks = data.transcript.map(t => t.words);
  const embeddings = await openai.embeddings.create({
    model: "text-embedding-3-small",
    input: chunks,
  });
  await qdrantUpsert(meetingId, {
    vectors: embeddings.data.map(e => e.embedding),
    payloads: chunks.map(c => ({ text: c })),
  });
  return NextResponse.json({ ok: true });
}
// GET /api/search?q=&meetingId=
// Semantic search across embeddings
export async function GET(req: Request) {
  const { q, meetingId } = parseParams(req);
  const [queryVec] = await embedText(q);
  const results = await qdrantClient.search(
    meetingId ?? "global",
    { vector: queryVec, limit: 5,
      with_payload: true }
  );
  return NextResponse.json(results);
}
// Drizzle schema (simplified)
export const meetings = pgTable("meetings", {
  id:          uuid().primaryKey().defaultRandom(),
  title:       text().notNull(),
  joinUrl:     text(),
  status:      text().default("pending"),
  botId:       text(),
  participants: text().array(),
  createdAt:   timestamp().defaultNow(),
});
